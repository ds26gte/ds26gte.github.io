The S-Exp Datatype

This datatype defines the result of parsing s-expressions. See read-s-exp for more details and examples.

S-Exp<()>:

   | s-list(exps :: List<S-Exp>)
   | s-num(n :: Number)
   | s-str(s :: String)
   | s-sym(s :: String)
end

s-list :: (exps :: contract) -> S-Exp

s-num :: (n :: contract) -> S-Exp

s-str :: (s :: contract) -> S-Exp

s-sym :: (s :: contract) -> S-Exp
is-s-list :: () -> return
is-s-num :: () -> return
is-s-str :: () -> return
is-s-sym :: () -> return
